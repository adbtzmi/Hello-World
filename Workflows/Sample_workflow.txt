from resultsManagerAPI import resultsManagerObject as CESI
from all_bins import CODE_PRIME, CODE_INIT
import startup
import TestRunner
import json
import traceback

class CSendGearmanRequest(TestRunner.TestRunner):

    ############################################################################
    def __init__(self):
        self.verbosity = None
        self.debug = None
        self.mid = None
        self.phase = 0  # Default to Phase 0 (active) to protect legacy products
        self.gearman_server = None
        self.job_name = None
        self.timeout = 300

    ############################################################################
    def run_test(self, params):
        """
        Main test execution method
        
        Phase behavior:
        - Phase 0: Execute PFOM BB comparison data collection (legacy/active)
        - Phase 1: Skip execution, return CODE_PRIME (7500/9550 DOE complete)
        - Phase 2: Reserved for future logic
        """
        (self.verbosity, self.debug, self.mid) = startup.startup_params_check(params)
        
        # Read phase parameter from test configuration
        # Default to 0 to ensure other products continue normal operation
        self.phase = int(params.get('phase', 0))
        
        # Log phase information
        CESI.logWrite(self.mid, "INFO", 
                     f"PFOM Data Collection Phase: {self.phase}")
        
        # Phase 1: Skip execution per JIRA requirement
        # This stops PFOM BB comparison for products where DOE is complete
        if self.phase == 1:
            CESI.logWrite(self.mid, "INFO", 
                         "Phase 1 Detected: Skipping PFOM BB Comparison data collection.")
            CESI.logWrite(self.mid, "INFO", 
                         "Reason: DOE sample collection complete for this product.")
            
            # Set passing result code
            self.result.value = CODE_PRIME
            
            # Add test attributes for tracking
            CESI.setTestAttr(self.mid, "PFOM_PHASE", str(self.phase))
            CESI.setTestAttr(self.mid, "PFOM_SKIPPED", "TRUE")
            CESI.setTestAttr(self.mid, "PFOM_SKIP_REASON", "Phase 1 - DOE Complete")
            
            return CODE_PRIME
        
        # Phase 0: Execute normal PFOM data collection
        elif self.phase == 0:
            CESI.logWrite(self.mid, "INFO", 
                         "Phase 0: Executing PFOM BB Comparison data collection.")
            return self._execute_pfom_collection(params)
        
        # Phase 2: Reserved for future implementation
        elif self.phase == 2:
            CESI.logWrite(self.mid, "INFO", 
                         "Phase 2: Executing future PFOM logic (if defined).")
            return self._execute_phase2_logic(params)
        
        # Unknown phase - log error and fail safe to Phase 0
        else:
            CESI.logWrite(self.mid, "WARNING", 
                         f"Unknown phase value: {self.phase}. Defaulting to Phase 0.")
            self.phase = 0
            return self._execute_pfom_collection(params)

    ############################################################################
    def _execute_pfom_collection(self, params):
        """
        Execute Phase 0 PFOM Bad Block comparison data collection
        Original legacy logic for active data collection
        """
        try:
            # Read parameters
            self.gearman_server = params.get('gearman_server', '')
            self.job_name = params.get('job_name', 'pfom_bb_compare')
            self.timeout = int(params.get('timeout', 300))
            
            # Collect drive data for Gearman request
            drive_data = self._collect_drive_data()
            
            # Send Gearman request
            result = self._send_gearman_request(drive_data)
            
            # Process response
            if result and result.get('status') == 'success':
                CESI.logWrite(self.mid, "INFO", 
                             "PFOM data collection completed successfully.")
                
                # Store comparison results
                compare_result = result.get('compare_bb_result', 'UNKNOWN')
                CESI.setTestAttr(self.mid, "PFOM_BB_COMPARE", compare_result)
                CESI.setTestAttr(self.mid, "PFOM_PHASE", str(self.phase))
                
                # Check if comparison passed
                if compare_result == 'PASS':
                    self.result.value = CODE_PRIME
                    return CODE_PRIME
                else:
                    # Comparison failed - this would have been REJECT
                    # but interim disposition may be HOLD
                    CESI.logWrite(self.mid, "WARNING", 
                                 f"PFOM BB Comparison failed: {compare_result}")
                    self.result.value = result.get('bin_code', 'PFOM_BB_FAIL')
                    return self.result.value
            else:
                error_msg = result.get('error', 'Unknown error') if result else 'No response'
                CESI.logWrite(self.mid, "ERROR", 
                             f"PFOM data collection failed: {error_msg}")
                self.result.value = 'PFOM_COMM_FAIL'
                return self.result.value
                
        except Exception as e:
            CESI.logWrite(self.mid, "ERROR", 
                         f"Exception in PFOM collection: {str(e)}")
            CESI.logWrite(self.mid, "ERROR", traceback.format_exc())
            self.result.value = 'PFOM_EXCEPTION'
            return self.result.value

    ############################################################################
    def _execute_phase2_logic(self, params):
        """

        """

        return self._execute_pfom_collection(params)

    def _collect_drive_data(self):
        """
        Collect drive attributes needed for PFOM analysis
        """
        drive_data = {}
        
        # Collect standard drive identifiers
        attrs_to_collect = [
            'MID', 'FID', 'PRODUCT', 'DENSITY', 'FORM_FACTOR',
            'FIRMWARE', 'CUSTOMER_ID', 'TEST_STEP', 'TESTER_ID'
        ]
        
        for attr in attrs_to_collect:
            value = CESI.getSessionAttrs(self.mid, attr).get(attr, '')
            drive_data[attr] = value
        
        # Collect bad block information
        bb_data = self._get_bad_block_data()
        drive_data['bad_blocks'] = bb_data
        
        return drive_data

    ############################################################################
    def _get_bad_block_data(self):
        """
        Extract bad block data from drive or test results
        """
        # Placeholder - implementation depends on how BB data is stored
        # This would typically read from test attributes or drive SMART data
        bb_data = {
            'erase_failures': [],
            'program_failures': [],
            'read_failures': []
        }
        return bb_data

    ############################################################################
    def _send_gearman_request(self, drive_data):
        """
        Send request to Gearman job server
        """
        # Placeholder for actual Gearman communication
        # Implementation would use Gearman client library
        try:
            import gearman
            
            gm_client = gearman.GearmanClient([self.gearman_server])
            
            # Convert data to JSON
            payload = json.dumps(drive_data)
            
            # Submit job
            job_request = gm_client.submit_job(
                self.job_name, 
                payload,
                wait_until_complete=True,
                max_retries=3
            )
            
            if job_request.complete:
                result = json.loads(job_request.result)
                return result
            else:
                return {'status': 'failed', 'error': 'Job did not complete'}
                
        except Exception as e:
            CESI.logWrite(self.mid, "ERROR", 
                         f"Gearman communication error: {str(e)}")
            return {'status': 'failed', 'error': str(e)}
